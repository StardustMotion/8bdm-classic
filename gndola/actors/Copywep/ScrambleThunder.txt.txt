actor ScrambleThunderBall
{
Translation "192:192=228:228","198:198=228:228"
PROJECTILE
damagetype "ScrambleThunder"
Obituary "$OB_SCRAMBLETHUNDER"
+RIPPER
Damage (1)
Speed 60
Radius 8
Height 8
Scale 2.5
States
{
Spawn:
TNT1 AA 0 
TNT1 A 0 A_JumpIf(momz>=0, "Up")
TNT1 A 0 A_JumpIf(momz<0, "Down")
goto PushDown
Up:
FSST XYZY 1 A_Changevelocity(0,0,momz*1.00125,CVF_RELATIVE)
loop
PushDown:
TNT1 A 0 ThrustThingZ(0,40,1,0)
Down:
FSSI XYZY 1 A_Changevelocity(0,0,momz*1.00125,CVF_RELATIVE)
loop
Fall:
TNT1 A 0 ThrustThingZ(0,40,1,0)
FallLOOP:
TNT1 A 0 A_Changevelocity(0,0,momz*1.00125,CVF_RELATIVE)
FSSI XYZY 1 A_CheckFloor("SplitFloor")
loop
FallBall:
TNT1 A 0 A_PlaySound("fuseman/hvoltage","Weapon")
TNT1 A 0 ThrustThingZ(0,40,1,0)
FallBallLOOP:
TNT1 A 0 A_Changevelocity(0,0,momz*1.00125,CVF_RELATIVE)
SCTH ABCD 1 A_CheckFloor("SplitFloor")
loop
Death:
SCTH A 0 A_CheckFloor("SplitFloor")
SCTH A 0 A_CheckCeiling("Fall")
goto FallBall
SplitFloor:
TNT1 A 0 A_PlaySound("fuseman/hvoltage","Weapon")
TNT1 A 0 A_SpawnItemEx("ScrambleThunderSplitFloor",0,0,0,50,0,0,45,1)
TNT1 A 0 A_SpawnItemEx("ScrambleThunderSplitFloor",0,0,0,50,0,0,135,1)
TNT1 A 0 A_SpawnItemEx("ScrambleThunderSplitFloor",0,0,0,50,0,0,225,1)
TNT1 A 0 A_SpawnItemEx("ScrambleThunderSplitFloor",0,0,0,50,0,0,315,1)
stop
SplitCeiling:
TNT1 A 0 A_CustomMissile("ScrambleThunderSplitCeiling",0,0,0,2,0)
TNT1 A 0 A_CustomMissile("ScrambleThunderSplitCeiling",0,0,90,2,0)
TNT1 A 0 A_CustomMissile("ScrambleThunderSplitCeiling",0,0,180,2,0)
TNT1 A 0 A_CustomMissile("ScrambleThunderSplitCeiling",0,0,270,2,0)
stop
}
}
